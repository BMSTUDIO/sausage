Prevent crazy amounts of scroll events from being fired by setting an interval and listening.

 
            setInterval(function(){
                
                if (!self.hasScrolled)
                {
                    return;
                }
                
                self.hasScrolled = true;
                
                var st = $(window).scrollTop(),
                    h_win = $(window).height(),
                    h_doc = $(document).height(),
                    i = Math.floor((st + h_win/2)/h_doc*self.count);
                
                self._update(i);
                
            }, 250);

        _delegates: function () {

                })
                .delegate('.sausage', 'click', function(e){
                    e.preventDefault();

                        o = self.$inner.find(self.options.page).eq(val).offset().top;
